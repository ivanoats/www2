def period
  case self.recurring_month
  when 12
    1.year # 1.year != 12.months
  else
    self.recurring_month.months
  end
end

def period_in_words
  case recurring_month
  when 12
    'yearly'
  when 1
    'monthly'
  when 0
    "once"
  else
    "every #{recurring_month} months"
  end
end
